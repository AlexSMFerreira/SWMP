# Dynamic map -> base_link from the live /nav solution (PoseBroadcasterNode). Also

# Static base_link -> camera_left_rect (lever arm). IDENTITY PLACEHOLDER until the
# GPS-antenna -> camera offset is measured (Phase 3 in NEXT_STEPS_POSE.md). The cloud
# is already remapped to a ROS body convention (X-fwd, Y-left, Z-up) in the pointcloud
# node, so this stays a pure translation/rotation once the lever arm is known.
ros2 run tf2_ros static_transform_publisher \
  0 0 0 0 0 0 \
  base_link camera_left_rect

# Static base_link -> rslidar (lidar mounting). IDENTITY PLACEHOLDER until the lidar
# offset/orientation on the airship is measured. Without this the /rslidar_points cloud
# (frame_id "rslidar") has no path to "map" and RViz shows nothing for it.
ros2 run tf2_ros static_transform_publisher \
  0 0 0 0 0 0 \
  base_link rslidar
